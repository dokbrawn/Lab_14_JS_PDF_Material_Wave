import { jsPDF } from "https://cdn.jsdelivr.net/npm/jspdf@2.5.1/+esm"; // Сохраняем изменения в localStorage document.querySelectorAll('[contenteditable]').forEach(el => { const key = el.className || el.id; el.innerHTML = localStorage.getItem(key) || el.innerHTML; el.addEventListener('input', () => { localStorage.setItem(key, el.innerHTML); }); }); // Скачивание PDF document.getElementById('download-pdf').addEventListener('click', () => { const content = document.querySelector('.resume-container'); const pdf = new jsPDF(); pdf.html(content, { callback: function (doc) { doc.save("resume.pdf"); }, x: 10, y: 10, width: 180 }); }); // Ripple-эффект document.querySelectorAll('[data-ripple]').forEach(el => { el.addEventListener('click', function (e) { const circle = document.createElement('span'); const diameter = Math.max(el.clientWidth, el.clientHeight); const radius = diameter / 2; circle.style.width = circle.style.height = `${diameter}px`; circle.style.left = `${e.clientX - el.getBoundingClientRect().left - radius}px`; circle.style.top = `${e.clientY - el.getBoundingClientRect().top - radius}px`; circle.classList.add('ripple'); const ripple = el.getElementsByClassName('ripple')[0]; if (ripple) ripple.remove(); el.appendChild(circle); }); });
